  <!-- Corrected AJAX Script -->
  <script>
    $(function () {
      var authToken = "Q7ESBsvkGAb1xt8ViNkdvw";
      var accountId = "7ed1d0c0-23a5-4947-827f-ec4872351b7f";

      $("#shopvox-contactForm-error-message").hide();

      $("#contactForm").submit(function (e) {
        e.preventDefault();

        var name = $("#primary_contact_name").val();
        var company = $("#company_name").val();
        var phone = $("#primaryPhone").val();
        var email = $("#email").val();

        if (!name || !company || !phone || !email) {
          alert("Please fill out all required fields.");
          return;
        }

        var robotProtectionValue = $("#vox_protection").val();
        if (robotProtectionValue !== "") {
          return;
        }

        var formData = new FormData(this);

        $.ajax({
          url: "https://api.shopvox.com/v1/sales_leads.json",
          type: "POST",
          data: formData,
          processData: false,
          contentType: false,
          headers: {
            "Authorization": "Bearer " + authToken,
            "account_id": accountId
          },
          success: function (data) {
            window.location.href = $("#on_success").val();
          },
          error: function (xhr, status, error) {
            let message = "Submission failed.";
            if (xhr.responseJSON && xhr.responseJSON.errors) {
              message += "<br>" + xhr.responseJSON.errors.join("<br>");
            }
            $("#shopvox-contactForm-error-message").html(message).show();
          }
        });
      });
    });
  </script>
